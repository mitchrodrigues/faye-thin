def subscribe
        begin
        @publisher.publish('/users/list', {
                :channel => @channel,
                :client  => @client,
                :data    => Client.get_list,
                :message => "Connect",
                :count   => Client.count,
        })
        rescue => e
                LOGGER.info e.message
                LOGGER.info e.backtrace.join("\n")
        end
end
